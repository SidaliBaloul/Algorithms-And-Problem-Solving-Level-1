enum enMonths {january = 1, fabruary = 2, march = 3, april = 4, may = 5, june = 6, july = 7, august = 8, sept = 9, oct = 10};

int ReadMonth(string msg, int from, int to)
{
    int month = 0;
    do
    {
        cout << msg << from << " to " << to << endl;
        cin >> month;
    } while (month < from || month > to);

    return month;
}

{
